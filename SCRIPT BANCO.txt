CREATE TABLE COLABORADOR(
ID_COLABORADOR INTEGER NOT NULL PRIMARY KEY AUTO_INCREMENT,
NOME_COLABORADOR VARCHAR(250) NOT NULL,
CPF VARCHAR(14) NOT NULL,
TELEFONE VARCHAR(18) NOT NULL,
SEXO VARCHAR(1) NOT NULL,
DATA_NASCIMENTO DATE NOT NULL
)

CREATE TABLE CLIENTE(
ID_CLIENTE INTEGER NOT NULL PRIMARY KEY AUTO_INCREMENT,
NOME_CLIENTE VARCHAR(250) NOT NULL,
CNPJ VARCHAR(18)NOT NULL,
TELEFONE VARCHAR(18) NOT NULL,
ENDERECO VARCHAR(250) NOT NULL,
NOME_CONTATO VARCHAR(250) NOT NULL
)

CREATE TABLE FAZENDA(
ID_FAZENDA INTEGER NOT NULL PRIMARY KEY AUTO_INCREMENT,
NOME_FAZENDA VARCHAR(250) NOT NULL,
CNPJ VARCHAR(18) NOT NULL,
TELEFONE VARCHAR(18) NOT NULL,
ENDERECO VARCHAR(250) NOT NULL,
TIPO_PASTO VARCHAR(250) NOT NULL,
TIPO_ANIMAL VARCHAR(250) NOT NULL,
NOME_CONTATO VARCHAR(250) NOT NULL
)

CREATE TABLE ANIMAL(
ID_ANIMAL INTEGER NOT NULL PRIMARY KEY AUTO_INCREMENT,
ID_FAZENDA INTEGER NOT NULL,
RACA_ANIMAL VARCHAR(250) NOT NULL,
ESPECIE_ANIMAL VARCHAR(250) NOT NULL,
PESO_ANIMAL DOUBLE NOT NULL,
SEXO VARCHAR(1) NOT NULL,
CONSTRAINT FK_FAZENDA_ANIMAL FOREIGN KEY(ID_FAZENDA) REFERENCES FAZENDA(ID_FAZENDA)
)

CREATE TABLE FRIGORIFICO(
ID_FRIGORIFICO INTEGER NOT NULL PRIMARY KEY AUTO_INCREMENT,
NOME_FRIGORIFICO VARCHAR(250) NOT NULL,
CNPJ VARCHAR(18) NOT NULL,
TELEFONE VARCHAR(18) NOT NULL,
ENDERECO VARCHAR(250) NOT NULL,
NOME_RESPONSAVEL VARCHAR(250) NOT NULL
)

CREATE TABLE PRODUTO(
ID_PRODUTO INTEGER NOT NULL PRIMARY KEY AUTO_INCREMENT,
ID_FRIGORIFICO INTEGER NOT NULL,
NOME_PRODUTO VARCHAR(250) NOT NULL,
DATA_ABATE DATE NOT NULL,
VALIDADE_PRODUTO DATE NOT NULL,
LOTE_PRODUTO VARCHAR(50) NOT NULL,
VALOR_VENDA DOUBLE NOT NULL,
PESO_TOTAL DOUBLE NOT NULL,
CONSTRAINT FK_PRODUTO_FRIGORIFICO FOREIGN KEY(ID_FRIGORIFICO) REFERENCES FRIGORIFICO(ID_FRIGORIFICO)
)

CREATE TABLE VENDA(
ID_VENDA INTEGER NOT NULL PRIMARY KEY AUTO_INCREMENT,
ID_PRODUTO INTEGER NOT NULL,
ID_CLIENTE INTEGER NOT NULL,
DATA_VENDA DATE NOT NULL,
PESO DOUBLE NOT NULL,
VALOR_TOTAL DOUBLE NOT NULL,
CONSTRAINT FK_VENDA_PRODUTO FOREIGN KEY(ID_PRODUTO) REFERENCES PRODUTO(ID_PRODUTO),
CONSTRAINT FK_VENDA_CLIENTE FOREIGN KEY(ID_CLIENTE) REFERENCES CLIENTE(ID_CLIENTE)
)
